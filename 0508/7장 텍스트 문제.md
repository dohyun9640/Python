5-1(335p):
score_dic = {}

print("학생 성적 입력 프로그램입니다. (종료하려면 이름에 '종료'를 입력하세요)")

while True:
    name = input("\n학생 이름: ")
    
    if name == '종료':
        break
        
    score_input = input(f"{name}의 성적을 띄어쓰기로 구분하여 입력 (예: 99 83 95): ")
    scores = [int(x) for x in score_input.split()]
    score_dic[name] = scores
    print(f"[{name}] 학생의 성적이 저장되었습니다.")

print("\n--- 전체 학생 평균 성적 ---")
for name, scores in score_dic.items():
    if len(scores) > 0:
        avg = sum(scores) / len(scores)
        print(f"{name}의 평균 성적 = {avg:.1f}")

6-1(343p): [너무 어려워 AI의 힘을 빌림]
s = input("문자열을 입력하시오: ")

left = 0
right = len(s) - 1

is_palindrome = True 

while left < right:
    if s[left] != s[right]:
        is_palindrome = False
        break
    left += 1
    right -= 1

if is_palindrome:
    print("회문입니다.")
else:
    print("회문이 아닙니다.")
