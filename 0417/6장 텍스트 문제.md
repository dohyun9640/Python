def print_board(board):
    for r in range(3):
        print(" " + board[r][0] + " | " + board[r][1] + " | " + board[r][2])
        if r != 2:
            print("---+---+---")
    print()

def player_turn(board):
    while True:
        x = int(input("다음 수의 x좌표(0~2)를 입력하시오: "))
        y = int(input("다음 수의 y좌표(0~2)를 입력하시오: "))

        if board[x][y] != ' ':
            print("잘못된 위치입니다. 비어있는 다른 곳을 선택해주세요.\n")
        else:
            board[x][y] = 'X'
            break

def computer_turn(board):
    for r in range(3):
        for j in range(3):
            if board[r][j] == ' ':
                board[r][j] = 'O'
                return
